15/// Represents a digest of `DIGEST_LEN` bytes.
16#[serde_as]
17#[derive(Hash, PartialEq, Eq, Clone, Serialize, Deserialize, Ord, PartialOrd, Copy)]
18pub struct Digest<const DIGEST_LEN: usize> {
19    #[serde_as(as = "[_; DIGEST_LEN]")]
20    pub digest: [u8; DIGEST_LEN],
21}
22
23impl<const DIGEST_LEN: usize> Digest<DIGEST_LEN> {
24    /// Create a new digest containing the given bytes
25    pub fn new(digest: [u8; DIGEST_LEN]) -> Self {
26        Digest { digest }
27    }
28
29    /// The size of this digest in bytes.
30    pub fn size(&self) -> usize {
31        DIGEST_LEN
32    }
33}
34
35impl<const DIGEST_LEN: usize> fmt::Debug for Digest<DIGEST_LEN> {
36    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> Result<(), fmt::Error> {
37        write!(f, "{}", Base64::encode(self.digest))
38    }
39}
40
41impl<const DIGEST_LEN: usize> fmt::Display for Digest<DIGEST_LEN> {
42    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> Result<(), fmt::Error> {
43        write!(f, "{}", Base64::encode(self.digest))
44    }
45}
46
47impl<const DIGEST_LEN: usize> AsRef<[u8]> for Digest<DIGEST_LEN> {
48    fn as_ref(&self) -> &[u8] {
49        self.digest.as_ref()
50    }
51}
52
53impl<const DIGEST_LEN: usize> From<Digest<DIGEST_LEN>> for [u8; DIGEST_LEN] {
54    fn from(digest: Digest<DIGEST_LEN>) -> Self {
55        digest.digest
56    }
57}

59/// Trait implemented by hash functions providing a output of fixed length
60pub trait HashFunction<const DIGEST_LENGTH: usize>: Default {
61    /// The length of this hash functions digests in bytes.
62    const OUTPUT_SIZE: usize = DIGEST_LENGTH;
63
64    /// Create a new hash function of the given type
65    fn new() -> Self {
66        Self::default()
67    }
68
69    /// Process the given data, and update the internal of the hash function.
70    fn update<Data: AsRef<[u8]>>(&mut self, data: Data);
71
72    /// Retrieve result and consume hash function.
73    fn finalize(self) -> Digest<DIGEST_LENGTH>;
74
75    /// Compute the digest of the given data and consume the hash function.
76    fn digest<Data: AsRef<[u8]>>(data: Data) -> Digest<DIGEST_LENGTH> {
77        let mut h = Self::default();
78        h.update(data);
79        h.finalize()
80    }
81
82    /// Compute a single digest from all slices in the iterator in order and consume the hash function.
83    fn digest_iterator<K: AsRef<[u8]>, I: Iterator<Item = K>>(iter: I) -> Digest<DIGEST_LENGTH> {
84        let mut h = Self::default();
85        iter.for_each(|item| h.update(item));
86        h.finalize()
87    }
88}

90/// This trait is implemented by all messages that can be hashed.
91pub trait Hash<const DIGEST_LEN: usize> {
92    /// The type of the digest when this is hashed.
93    type TypedDigest: Into<Digest<DIGEST_LEN>> + Eq + std::hash::Hash + Copy + fmt::Debug;
94
95    fn digest(&self) -> Self::TypedDigest;
96}

98/// This wraps a [digest::Digest] as a [HashFunction].
99#[derive(Default)]
100pub struct HashFunctionWrapper<Variant, const DIGEST_LEN: usize>(Variant);
101
102impl<Variant: digest::Digest + Default, const DIGEST_LEN: usize> HashFunction<DIGEST_LEN>
103    for HashFunctionWrapper<Variant, DIGEST_LEN>
104{
105    fn update<Data: AsRef<[u8]>>(&mut self, data: Data) {
106        self.0.update(data);
107    }
108
109    fn finalize(self) -> Digest<DIGEST_LEN> {
110        let mut digest = [0u8; DIGEST_LEN];
111        self.0.finalize_into(GenericArray::from_mut_slice(&mut digest));
112        Digest { digest }
113    }
114}

116// Impl std::io::Write for HashFunctionWrapper. Needed for compatibility in Sui.
117impl<Variant: digest::Digest + Default, const DIGEST_LEN: usize> std::io::Write
118    for HashFunctionWrapper<Variant, DIGEST_LEN>
119{
120    fn write(&mut self, buf: &[u8]) -> std::io::Result<usize> {
121        self.update(buf);
122        Ok(buf.len())
123    }
124
125    fn flush(&mut self) -> std::io::Result<()> {
126        Ok(())
127    }
128}
