23/// Provides access to coin-related data such as coins owned by an address,
24/// balances, or metadata.
25#[rpc(client, namespace = "iotax")]
26pub trait CoinReadApi {
27    // /// Return all Coin<`coin_type`> objects owned by an address.
28    #[rustfmt::skip]
29    #[method(name = "getCoins")]
30    async fn get_coins(
31        &self,
32        // the owner's IOTA address
33        owner: IotaAddress,
34        // optional type name for the coin (e.g., 0x168da5bf1f48dafc111b0a488fa454aca95e0b5e::usdc::USDC), default to 0x2::iota::IOTA if not specified.
35        coin_type: Option<String>,
36        // optional paging cursor
37        cursor: Option<ObjectID>,
38        // maximum number of items per page
39        limit: Option<usize>,
40    ) -> RpcResult<CoinPage>;
41
42    /// Return the total coin balance for one coin type, owned by the address owner.
43    #[rustfmt::skip]
44    #[method(name = "getBalance")]
45    async fn get_balance(
46        &self,
47        // the owner's IOTA address
48        owner: IotaAddress,
49        // optional type names for the coin (e.g., 0x168da5bf1f48dafc111b0a488fa454aca95e0b5e::usdc::USDC), default to 0x2::iota::IOTA if not specified.
50        coin_type: Option<String>,
51    ) -> RpcResult<Balance>;
52}
53
54/// `next_cursor` points to the last item in the page;
55/// Reading with `next_cursor` will start from the next item after `next_cursor`
56/// if `next_cursor` is `Some`, otherwise it will start from the first item.
57#[derive(Clone, Debug, Deserialize, PartialEq, Eq)]
58#[serde(rename_all = "camelCase")]
59pub struct Page<T, C> {
60    pub data: Vec<T>,
61    pub next_cursor: Option<C>,
62    pub has_next_page: bool,
63}
64
65pub type CoinPage = Page<Coin, ObjectID>;
66
67#[serde_as]
68#[derive(Deserialize, Debug, PartialEq, Eq, Clone)]
69#[serde(rename_all = "camelCase")]
70pub struct Coin {
71    pub coin_type: String,
72    pub coin_object_id: ObjectID,
73    #[serde_as(as = "AsSequenceNumber")]
74    pub version: SequenceNumber,
75    pub digest: ObjectDigest,
76    #[serde_as(as = "BigInt<u64>")]
77    pub balance: u64,
78    pub previous_transaction: TransactionDigest,
79}
