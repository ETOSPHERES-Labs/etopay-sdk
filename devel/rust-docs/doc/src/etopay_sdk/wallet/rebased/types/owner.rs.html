1use super::{IotaAddress, SequenceNumber};

4#[derive(Eq, PartialEq, Debug, Clone, Copy, Deserialize, Serialize, Hash, Ord, PartialOrd)]
5#[allow(clippy::enum_variant_names)]
6pub enum Owner {
7    /// Object is exclusively owned by a single address, and is mutable.
8    AddressOwner(IotaAddress),
9    /// Object is exclusively owned by a single object, and is mutable.
10    /// The object ID is converted to IotaAddress as IotaAddress is universal.
11    ObjectOwner(IotaAddress),
12    /// Object is shared, can be used by any address, and is mutable.
13    Shared {
14        /// The version at which the object became shared
15        initial_shared_version: SequenceNumber,
16    },
17    /// Object is immutable, and hence ownership doesn't matter.
18    Immutable,
19}
