6use serde::{Deserialize, Serialize};

8use super::{Identifier, ObjectID, ObjectRef, SequenceNumber, TypeTag};

10/// A series of commands where the results of one command can be used in future
11/// commands
12#[derive(Debug, PartialEq, Eq, Hash, Clone, Serialize, Deserialize)]
13pub struct ProgrammableTransaction {
14    /// Input objects or primitive values
15    pub inputs: Vec<CallArg>,
16    /// The commands to be executed sequentially. A failure in any command will
17    /// result in the failure of the entire transaction.
18    pub commands: Vec<Command>,
19}
20
21/// A single command in a programmable transaction.
22#[derive(Debug, PartialEq, Eq, Hash, Clone, Serialize, Deserialize)]
23pub enum Command {
24    /// A call to either an entry or a public Move function
25    MoveCall(Box<ProgrammableMoveCall>),
26    /// `(Vec<forall T:key+store. T>, address)`
27    /// It sends n-objects to the specified address. These objects must have
28    /// store (public transfer) and either the previous owner must be an
29    /// address or the object must be newly created.
30    TransferObjects(Vec<Argument>, Argument),
31    /// `(&mut Coin<T>, Vec<u64>)` -> `Vec<Coin<T>>`
32    /// It splits off some amounts into a new coins with those amounts
33    SplitCoins(Argument, Vec<Argument>),
34    /// `(&mut Coin<T>, Vec<Coin<T>>)`
35    /// It merges n-coins into the first coin
36    MergeCoins(Argument, Vec<Argument>),
37    /// Publishes a Move package. It takes the package bytes and a list of the
38    /// package's transitive dependencies to link against on-chain.
39    Publish(Vec<Vec<u8>>, Vec<ObjectID>),
40    /// `forall T: Vec<T> -> vector<T>`
41    /// Given n-values of the same type, it constructs a vector. For non objects
42    /// or an empty vector, the type tag must be specified.
43    MakeMoveVec(Option<TypeTag>, Vec<Argument>),
44    /// Upgrades a Move package
45    /// Takes (in order):
46    /// 1. A vector of serialized modules for the package.
47    /// 2. A vector of object ids for the transitive dependencies of the new
48    ///    package.
49    /// 3. The object ID of the package being upgraded.
50    /// 4. An argument holding the `UpgradeTicket` that must have been produced
51    ///    from an earlier command in the same programmable transaction.
52    Upgrade(Vec<Vec<u8>>, Vec<ObjectID>, ObjectID, Argument),
53}
54
55impl Command {
56    pub fn move_call(
57        package: ObjectID,
58        module: Identifier,
59        function: Identifier,
60        type_arguments: Vec<TypeTag>,
61        arguments: Vec<Argument>,
62    ) -> Self {
63        Command::MoveCall(Box::new(ProgrammableMoveCall {
64            package,
65            module,
66            function,
67            type_arguments,
68            arguments,
69        }))
70    }
71}
72
73/// An argument to a programmable transaction command
74#[derive(Debug, PartialEq, Eq, Hash, Clone, Copy, Serialize, Deserialize)]
75pub enum Argument {
76    /// The gas coin. The gas coin can only be used by-ref, except for with
77    /// `TransferObjects`, which can use it by-value.
78    GasCoin,
79    /// One of the input objects or primitive values (from
80    /// `ProgrammableTransaction` inputs)
81    Input(u16),
82    /// The result of another command (from `ProgrammableTransaction` commands)
83    Result(u16),
84    /// Like a `Result` but it accesses a nested result. Currently, the only
85    /// usage of this is to access a value from a Move call with multiple
86    /// return values.
87    NestedResult(u16, u16),
88}
89
90/// The command for calling a Move function, either an entry function or a
91/// public function (which cannot return references).
92#[derive(Debug, PartialEq, Eq, Hash, Clone, Serialize, Deserialize)]
93pub struct ProgrammableMoveCall {
94    /// The package containing the module and function.
95    pub package: ObjectID,
96    /// The specific module in the package containing the function.
97    pub module: Identifier,
98    /// The function to be called.
99    pub function: Identifier,
100    /// The type arguments to the function.
101    pub type_arguments: Vec<TypeTag>,
102    /// The arguments to the function.
103    pub arguments: Vec<Argument>,
104}
105
106#[derive(Debug, PartialEq, Eq, Hash, Clone, Serialize, Deserialize)]
107pub enum CallArg {
108    // contains no structs or objects
109    Pure(Vec<u8>),
110    // an object
111    Object(ObjectArg),
112}
113
114#[derive(Debug, PartialEq, Eq, Hash, Clone, Copy, Serialize, Deserialize)]
115pub enum ObjectArg {
116    // A Move object, either immutable, or owned mutable.
117    ImmOrOwnedObject(ObjectRef),
118    // A Move object that's shared.
119    // SharedObject::mutable controls whether caller asks for a mutable reference to shared
120    // object.
121    SharedObject {
122        id: ObjectID,
123        initial_shared_version: SequenceNumber,
124        mutable: bool,
125    },
126    // A Move object that can be received in this transaction.
127    Receiving(ObjectRef),
128}
129
130impl ObjectArg {
131    // pub const IOTA_SYSTEM_MUT: Self = Self::SharedObject {
132    //     id: IOTA_SYSTEM_STATE_OBJECT_ID,
133    //     initial_shared_version: IOTA_SYSTEM_STATE_OBJECT_SHARED_VERSION,
134    //     mutable: true,
135    // };
136
137    pub fn id(&self) -> ObjectID {
138        match self {
139            ObjectArg::Receiving((id, _, _))
140            | ObjectArg::ImmOrOwnedObject((id, _, _))
141            | ObjectArg::SharedObject { id, .. } => *id,
142        }
143    }
144}
