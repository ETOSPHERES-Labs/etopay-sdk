18/// Trait impl'd by a key/keypair that can create signatures.
19///
20pub trait Signer<Sig> {
21    /// Create a new signature over a message.
22    fn sign(&self, msg: &[u8]) -> Sig;
23}
24
25/// The length of a private key in bytes.
26pub const ED25519_PRIVATE_KEY_LENGTH: usize = 32;
27
28/// The length of a public key in bytes.
29pub const ED25519_PUBLIC_KEY_LENGTH: usize = 32;
30
31/// The length of a signature in bytes.
32pub const ED25519_SIGNATURE_LENGTH: usize = 64;
33
34/// The key pair bytes length is the same as the private key length. This enforces deserialization to always derive the public key from the private key.
35pub const ED25519_KEYPAIR_LENGTH: usize = ED25519_PRIVATE_KEY_LENGTH;
36
37/// Ed25519 public key.
38#[derive(Clone, PartialEq, Eq)]
39pub struct Ed25519PublicKey(pub ed25519_consensus::VerificationKey);
40
41/// Ed25519 private key.
42#[derive(zeroize::ZeroizeOnDrop)]
43pub struct Ed25519PrivateKey(pub ed25519_consensus::SigningKey);
44
45/// Ed25519 key pair.
46#[derive(Debug, PartialEq, Eq)]
47pub struct Ed25519KeyPair {
48    public: Ed25519PublicKey,
49    private: Ed25519PrivateKey,
50}
51
52/// Ed25519 signature.
53#[derive(Debug, Clone)]
54pub struct Ed25519Signature {
55    pub sig: ed25519_consensus::Signature,
56    // Helps implementing AsRef<[u8]>.
57    pub bytes: OnceCell<[u8; ED25519_SIGNATURE_LENGTH]>,
58}
59
60impl Ed25519Signature {
61    pub const LENGTH: usize = ED25519_SIGNATURE_LENGTH;
62}
63
64impl AsRef<[u8]> for Ed25519Signature {
65    fn as_ref(&self) -> &[u8] {
66        self.bytes.get_or_init::<_>(|| self.sig.to_bytes())
67    }
68}
69
70//
71// Implementation of [Ed25519PrivateKey].
72//
73
74impl AsRef<[u8]> for Ed25519PrivateKey {
75    fn as_ref(&self) -> &[u8] {
76        self.0.as_ref()
77    }
78}
79
80impl PartialEq for Ed25519PrivateKey {
81    fn eq(&self, other: &Self) -> bool {
82        self.as_ref() == other.as_ref()
83    }
84}
85
86impl Eq for Ed25519PrivateKey {}
87
88impl fmt::Display for Ed25519PrivateKey {
89    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
90        write!(f, "Ed25519PrivateKey[]")
91    }
92}
93impl fmt::Debug for Ed25519PrivateKey {
94    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
95        write!(f, "Ed25519PrivateKey[]")
96    }
97}

105impl ToFromBytes for Ed25519PrivateKey {
106    fn from_bytes(bytes: &[u8]) -> Result<Self, RebasedError> {
107        ed25519_consensus::SigningKey::try_from(bytes)
108            .map(Ed25519PrivateKey)
109            .map_err(|_| RebasedError::InvalidInput)
110    }
111}

128impl AsRef<[u8]> for Ed25519PublicKey {
129    fn as_ref(&self) -> &[u8] {
130        self.0.as_ref()
131    }

134impl<'a> From<&'a Ed25519PrivateKey> for Ed25519PublicKey {
135    fn from(private: &'a Ed25519PrivateKey) -> Self {
136        Ed25519PublicKey(private.0.verification_key())
137    }
138}
139
140//
141// Implementation of [Ed25519KeyPair].
142//
143
144impl Ed25519KeyPair {
145    pub fn public(&'_ self) -> &'_ Ed25519PublicKey {
146        &self.public
147    }
148
149    pub fn generate<R: CryptoRng + RngCore>(rng: &mut R) -> Self {
150        let kp = ed25519_consensus::SigningKey::new(rng);
151        Ed25519KeyPair {
152            public: Ed25519PublicKey(kp.verification_key()),
153            private: Ed25519PrivateKey(kp),
154        }
155    }
156}
157
158impl From<Ed25519PrivateKey> for Ed25519KeyPair {
159    fn from(private: Ed25519PrivateKey) -> Self {
160        let public = Ed25519PublicKey::from(&private);
161        Ed25519KeyPair { public, private }
162    }
163}
164
165impl From<ed25519_consensus::SigningKey> for Ed25519KeyPair {
166    fn from(kp: ed25519_consensus::SigningKey) -> Self {
167        Ed25519KeyPair {
168            public: Ed25519PublicKey(kp.verification_key()),
169            private: Ed25519PrivateKey(kp),
170        }
171    }
172}
173
174impl Signer<Ed25519Signature> for Ed25519KeyPair {
175    fn sign(&self, msg: &[u8]) -> Ed25519Signature {
176        Ed25519Signature {
177            sig: self.private.0.sign(msg),
178            bytes: OnceCell::new(),
179        }
180    }
181}
