51/// Convert a [`u128`] to [`CryptoAmount`] while taking the decimals into account.
52fn convert_u128_to_crypto_amount(value: u128, decimals: u32) -> Result<CryptoAmount> {
53    let Some(mut result_decimal) = Decimal::from_u128(value) else {
54        return Err(WalletError::ConversionError(format!(
55            "could not convert u128 to Decimal: {value:?}"
56        )));
57    };
58
59    // directly set the decimals
60    result_decimal
61        .set_scale(decimals)
62        .map_err(|e| WalletError::ConversionError(format!("could not set scale to decimals: {e:?}")))?;
63
64    result_decimal.normalize_assign(); // remove trailing zeros
65
66    CryptoAmount::try_from(result_decimal).map_err(|e| {
67        WalletError::ConversionError(format!(
68            "could not convert decimal {result_decimal:?} to crypto amount: {e:?}"
69        ))
70    })
71}
72/// Convert a [`CryptoAmount`] to [`U256`] while taking the decimals into account.
73fn convert_crypto_amount_to_u128(amount: CryptoAmount, decimals: u32) -> Result<u128> {
74    // normalize to remove trailing zeros
75    let value_decimal = amount.inner().normalize();
76
77    let scale = value_decimal.scale();
78
79    // if the Decimal has more decimals than we will store in the U256, we cannot accurately represent this value.
80    if scale > decimals {
81        return Err(WalletError::ConversionError(format!(
82            "cannot represent value of {} in a U256 with {} decimals.",
83            value_decimal, decimals
84        )));
85    }
86
87    if value_decimal.is_sign_negative() {
88        return Err(WalletError::ConversionError(format!(
89            "cannot represent negative values: {}",
90            value_decimal
91        )));
92    }
93
94    // create a i128 from all the mantissa bits, then we just need to multiply by 10^(decimals - scale) to get the scaled value
95    let mantissa = value_decimal.mantissa() as u128; // we checked that it is not negative
96
97    // the scale is 10^(decimals-scale). Since we checked for scale > decimals above, (decimals - scale) >= 0
98    let exponent = decimals - scale;
99    let scale = 10u128
100        .checked_pow(exponent)
101        .ok_or_else(|| WalletError::ConversionError(format!("10^{exponent} does not fit in u128")))?;
102
103    let value = mantissa.checked_mul(scale).ok_or_else(|| {
104        WalletError::ConversionError(format!("result does not fit in U256: {} * {}", mantissa, scale))
105    })?;
106
107    Ok(value)
108}

130    async fn send_amount(&self, intent: &TransactionIntent) -> Result<String> {
131        let address = self.keystore.addresses()[0];
132
133        let recipient = intent.address_to.parse::<rebased::IotaAddress>()?;
134
135        // TODO: actually check to make sure the u64 can handle the u128 value
136        let amount = convert_crypto_amount_to_u128(intent.amount, self.decimals)? as u64;
137
138        let gas_budget = 5_000_000;
139
140        let coins = self
141            .client
142            .get_coins(address, Some(self.coin_type.clone()), None, None)
143            .await
144            .map_err(RebasedError::RpcError)?
145            .data;
146
147        // for now we just select _a_ coin object with enough balance, but at some point we probably need
148        // to automatically merge multiple objects into one to send them
149
150        let (pt, gas_coin) = if let Some(gas_coin) = coins.iter().find(|c| c.balance > (amount + gas_budget)) {
151            log::info!("Single coin to cover gas and transaction found: {gas_coin:?}");
152
153            let mut b = ProgrammableTransactionBuilder::new();
154
155            // provide the inputs
156            let input_amount = b.pure(amount).map_err(RebasedError::BuilderError)?;
157            let input_receiver = b.pure(recipient).map_err(RebasedError::BuilderError)?;
158
159            // split the gas coin depending on the amount to send
160            let Argument::Result(split_primary) = b.command(Command::SplitCoins(Argument::GasCoin, vec![input_amount]))
161            else {
162                panic!("self.command should always give a Argument::Result")
163            };
164
165            // actually transfer the object that resulted from the split
166            b.command(Command::TransferObjects(
167                vec![Argument::NestedResult(split_primary, 0)],
168                input_receiver,
169            ));
170
171            let pt = b.finish();
172            (pt, gas_coin)
173        } else {
174            // we do not have a single coin to cover amount + gas budget. Try to merge multiple
175            // coins until we have enough.
176
177            // coins[0];
178
179            return Err(WalletError::InsufficientBalance(String::new()));
180        };
181
182        // create the object ref manually instead of fetching as in the official sdk
183        let gas_coin_ref: rebased::ObjectRef = (gas_coin.coin_object_id, gas_coin.version, gas_coin.digest);
184
185        let gas_price = self
186            .client
187            .get_reference_gas_price()
188            .await
189            .map_err(RebasedError::RpcError)?;
190
191        let tx_data = rebased::TransactionData::V1(rebased::TransactionDataV1 {
192            kind: TransactionKind::ProgrammableTransaction(pt),
193            sender: address,
194            gas_data: GasData {
195                payment: vec![gas_coin_ref],
196                owner: address,
197                price: *gas_price,
198                budget: gas_budget,
199            },
200            expiration: TransactionExpiration::None,
201        });
202
203        let signature = self
204            .keystore
205            .sign_secure(&address, &tx_data, rebased::Intent::iota_transaction())?;
206
207        let tx = rebased::Transaction::from_data(tx_data, vec![signature.clone()]);
208
209        let (tx_bytes, signatures) = tx.to_tx_bytes_and_signatures()?;
210
211        let transaction_block_response = self
212            .client
213            .execute_transaction_block(
214                tx_bytes.clone(),
215                signatures.clone(),
216                Some(rebased::IotaTransactionBlockResponseOptions::full_content()),
217                Some(rebased::ExecuteTransactionRequestType::WaitForLocalExecution),
218            )
219            .await
220            .map_err(RebasedError::RpcError)?;
221
222        log::info!("Transaction sent {}", transaction_block_response.digest);
223        log::info!("Response:\n{transaction_block_response:?}");
224
225        if !transaction_block_response.errors.is_empty() {
226            log::warn!("Errors: {:?}", transaction_block_response.errors);
227        }
228
229        Ok(transaction_block_response.digest.to_string())
230    }
