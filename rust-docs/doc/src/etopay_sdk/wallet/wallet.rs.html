22/// Wallet user interface
23pub trait WalletUser: Debug {
24    /// Gets a new address for the user
25    ///
26    /// # Returns
27    ///
28    /// Returns a `Result` containing the generated address as a `String` if successful, or an `Error` if it fails.
29    ///
30    /// # Errors
31    ///
32    /// This function can return an error if it fails to synchronize the wallet, generate addresses, or encounter any other issues.
33    async fn get_address(&self) -> Result<String>;
34
35    /// Gets the balance of a user.
36    ///
37    /// # Returns
38    ///
39    /// Returns the available balance of the user as a `f64` if successful, or an `Error` if it fails.
40    ///
41    /// # Errors
42    ///
43    /// This function can return an error if it fails to synchronize the wallet or encounters any other issues.
44    async fn get_balance(&self) -> Result<CryptoAmount>;
45
46    /// Send amount to receiver
47    ///
48    /// # Arguments
49    ///
50    /// * `address` - The address of the receiver.
51    /// * `amount` - The amount to send.
52    /// * `tag` - The transactions tag. Optional.
53    /// * `message` - The transactions message. Optional.
54    ///
55    ///
56    /// Returns a `Result` containing the sent transaction ID if successful, or an `Error` if it fails.
57    ///
58    /// # Errors
59    ///
60    /// This function can return an error if it fails to synchronize the wallet, send the transaction, or encounter any other issues.
61    async fn send_amount(&self, intent: &TransactionIntent) -> Result<String>;
62
63    /// Gets the list of transactions
64    ///
65    /// # Arguments
66    ///
67    /// * `start` - The index of the first wallet transaction to return
68    /// * `limit` - The number of following wallet transactions to return
69    ///
70    /// # Returns
71    ///
72    /// The list of wallet transactions.
73    ///
74    /// # Errors
75    ///
76    /// This function can return an error if it cannot retrieve the list of wallet transactions.
77    async fn get_wallet_tx_list(&self, start: usize, limit: usize) -> Result<WalletTxInfoList>;
78
79    /// Get detailed report of a particular transaction in the history
80    ///
81    /// # Arguments
82    ///
83    /// * `tx_id` - The id of the wallet transaction to return the details for.
84    ///
85    /// # Returns
86    ///
87    /// The wallet transaction details.
88    ///
89    /// # Errors
90    ///
91    /// This function can return an error if it cannot retrieve the wallet transaction.
92    async fn get_wallet_tx(&self, tx_id: &str) -> Result<WalletTxInfo>;
93
94    /// Estimate gas cost for eip 1559 transaction
95    ///
96    /// # Arguments
97    ///
98    /// * `transaction` - A transaction with a priority fee ([EIP-1559](https://eips.ethereum.org/EIPS/eip-1559))
99    ///
100    /// # Returns the estimated gas cost for the underlying transaction to be executed (gas limit, max fee per gas and max priority fee per gas)
101    ///
102    /// This function can return an error if it cannot parse input transaction or retrieve information from the node.
103    async fn estimate_gas_cost(&self, intent: &TransactionIntent) -> Result<GasCostEstimation>;
104}
