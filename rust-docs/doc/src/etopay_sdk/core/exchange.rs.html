8impl Sdk {
9    /// Return the current exchange rate.
10    ///
11    /// # Returns
12    ///
13    /// Returns a `Result` containing the exchange rate as a `Decimal` type if successful, or a [`crate::Error`] if an error occurs.
14    // MARK10:get_exchange_rate
15    pub async fn get_exchange_rate(&self) -> Result<Decimal> {
16        info!("Fetching exchange rate from viviswap");
17        let _user = self.get_user().await?;
18
19        let access_token = self
20            .access_token
21            .as_ref()
22            .ok_or(crate::error::Error::MissingAccessToken)?;
23        let config = self.config.as_ref().ok_or(crate::Error::MissingConfig)?;
24        let network = self.active_network.clone().ok_or(crate::Error::MissingNetwork)?;
25        let currency = Currency::try_from(network.display_symbol)?;
26        let exchange_rate = get_viviswap_exchange_rate(config, access_token, currency).await?;
27        Ok(exchange_rate)
28    }
29}
