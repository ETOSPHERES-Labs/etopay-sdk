67#[derive(Eq, PartialEq, Clone, Debug, Serialize, Deserialize)]
68#[serde(rename = "ExecutionStatus", rename_all = "camelCase", tag = "status")]
69pub enum IotaExecutionStatus {
70    // Gas used in the success case.
71    Success,
72    // Gas used in the failed case, and the error.
73    Failure { error: String },
74}
75
76impl Display for IotaExecutionStatus {
77    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
78        match self {
79            Self::Success => write!(f, "success"),
80            Self::Failure { error } => write!(f, "failure due to {error}"),
81        }
82    }
83}
84
85impl IotaExecutionStatus {
86    pub fn is_ok(&self) -> bool {
87        matches!(self, IotaExecutionStatus::Success)
88    }
89    pub fn is_err(&self) -> bool {
90        matches!(self, IotaExecutionStatus::Failure { .. })
91    }
92}
