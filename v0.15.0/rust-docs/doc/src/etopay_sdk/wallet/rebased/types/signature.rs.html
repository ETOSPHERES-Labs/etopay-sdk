21#[derive(Clone, Copy, Deserialize, Serialize, Debug, PartialEq, Eq)]
22pub enum SignatureScheme {
23    ED25519,
24    // Secp256k1,
25    // Secp256r1,
26    // BLS12381, // This is currently not supported for user Iota Address.
27    // MultiSig,
28    // ZkLoginAuthenticator,
29    // PasskeyAuthenticator,
30}
31
32impl SignatureScheme {
33    pub fn flag(&self) -> u8 {
34        match self {
35            SignatureScheme::ED25519 => 0x00,
36            // SignatureScheme::Secp256k1 => 0x01,
37            // SignatureScheme::Secp256r1 => 0x02,
38            // SignatureScheme::MultiSig => 0x03,
39            // SignatureScheme::BLS12381 => 0x04, // This is currently not supported for user Iota
40            // // Address.
41            // SignatureScheme::ZkLoginAuthenticator => 0x05,
42            // SignatureScheme::PasskeyAuthenticator => 0x06,
43        }
44    }
45
46    /// Takes as input an hasher and updates it with a flag byte if the input
47    /// scheme is not ED25519; it does nothing otherwise.
48    pub fn update_hasher_with_flag(&self, hasher: &mut Blake2b256) {
49        #[allow(unreachable_patterns)]
50        match self {
51            SignatureScheme::ED25519 => (),
52            _ => hasher.update([self.flag()]),
53        };
54    }
55
56    pub fn from_flag(flag: &str) -> Result<SignatureScheme, RebasedError> {
57        let byte_int = flag
58            .parse::<u8>()
59            .map_err(|_| RebasedError::KeyConversion("Invalid key scheme".to_string()))?;
60        Self::from_flag_byte(&byte_int)
61    }
62
63    pub fn from_flag_byte(byte_int: &u8) -> Result<SignatureScheme, RebasedError> {
64        match byte_int {
65            0x00 => Ok(SignatureScheme::ED25519),
66            // 0x01 => Ok(SignatureScheme::Secp256k1),
67            // 0x02 => Ok(SignatureScheme::Secp256r1),
68            // 0x03 => Ok(SignatureScheme::MultiSig),
69            // 0x04 => Ok(SignatureScheme::BLS12381),
70            // 0x05 => Ok(SignatureScheme::ZkLoginAuthenticator),
71            // 0x06 => Ok(SignatureScheme::PasskeyAuthenticator),
72            _ => Err(RebasedError::KeyConversion("Invalid key scheme".to_string())),
73        }
74    }
75}

82impl Serialize for Signature {
83    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
84    where
85        S: Serializer,
86    {
87        let bytes = self.as_ref();
88
89        if serializer.is_human_readable() {
90            let s = Base64::encode(bytes);
91            serializer.serialize_str(&s)
92        } else {
93            serializer.serialize_bytes(bytes)
94        }
95    }
96}
97
98impl<'de> Deserialize<'de> for Signature {
99    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
100    where
101        D: Deserializer<'de>,
102    {
103        use serde::de::Error;
104
105        let bytes = if deserializer.is_human_readable() {
106            let s = String::deserialize(deserializer)?;
107            Base64::decode(&s).map_err(|e| Error::custom(e.to_string()))?
108        } else {
109            let data: Vec<u8> = Vec::deserialize(deserializer)?;
110            data
111        };
112
113        Self::from_bytes(&bytes).map_err(|e| Error::custom(e.to_string()))
114    }
115}

117impl Signature {
118    /// The messaged passed in is already hashed form.
119    pub fn new_hashed(hashed_msg: &[u8], secret: &dyn Signer<Signature>) -> Self {
120        Signer::sign(secret, hashed_msg)
121    }
122
123    pub fn new_secure<T>(value: &IntentMessage<T>, secret: &dyn Signer<Signature>) -> Result<Self, RebasedError>
124    where
125        T: Serialize,
126    {
127        // Compute the BCS hash of the value in intent message. In the case of
128        // transaction data, this is the BCS hash of `struct TransactionData`,
129        // different from the transaction digest itself that computes the BCS
130        // hash of the Rust type prefix and `struct TransactionData`.
131        // (See `fn digest` in `impl Message for SenderSignedData`).
132        let mut hasher = Blake2b256::default();
133        hasher.update(bcs::to_bytes(&value)?);
134
135        Ok(Signer::sign(secret, &hasher.finalize().digest))
136    }
137}

221impl ToFromBytes for Ed25519IotaSignature {
222    fn from_bytes(bytes: &[u8]) -> Result<Self, RebasedError> {
223        if bytes.len() != Self::LENGTH {
224            return Err(RebasedError::InputLengthWrong(Self::LENGTH));
225        }
226        let mut sig_bytes = [0; Self::LENGTH];
227        sig_bytes.copy_from_slice(bytes);
228        Ok(Self(sig_bytes))
229    }
230}

232impl Signer<Signature> for Ed25519KeyPair {
233    fn sign(&self, msg: &[u8]) -> Signature {
234        let sig: Ed25519Signature = <Self as Signer<Ed25519Signature>>::sign(self, msg);
235
236        let mut signature_bytes: Vec<u8> = Vec::new();
237        signature_bytes.extend_from_slice(&[Ed25519IotaSignature::SCHEME.flag()]);
238        signature_bytes.extend_from_slice(sig.as_ref());
239        signature_bytes.extend_from_slice(self.public().as_ref());
240        #[allow(
241            clippy::expect_used,
242            reason = "the required length is constant, thus this is acceptable"
243        )]
244        let sign = Ed25519IotaSignature::from_bytes(&signature_bytes[..])
245            .expect("Serialized signature did not have expected size");
246
247        Signature::Ed25519IotaSignature(sign)
248    }

251/// Due to the incompatibility of [enum Signature] (which dispatches a trait
252/// that assumes signature and pubkey bytes for verification), here we add a
253/// wrapper enum where member can just implement a lightweight [trait
254/// AuthenticatorTrait]. This way MultiSig (and future Authenticators) can
255/// implement its own `verify`.
256#[derive(Debug, Clone, PartialEq, Eq, Hash)]
257pub enum GenericSignature {
258    // MultiSig,
259    Signature(Signature),
260    // ZkLoginAuthenticator,
261    // PasskeyAuthenticator,
262}
263
264impl From<Signature> for GenericSignature {
265    fn from(value: Signature) -> Self {
266        Self::Signature(value)
267    }
268}

270/// GenericSignature encodes a single signature [enum Signature] as is `flag ||
271/// signature || pubkey`. It encodes [struct MultiSigLegacy] as the MultiSig
272/// flag (0x03) concat with the bcs serializedbytes of [struct MultiSigLegacy]
273/// i.e. `flag || bcs_bytes(MultiSigLegacy)`. [struct Multisig] is encodede as
274/// the MultiSig flag (0x03) concat with the bcs serializedbytes of [struct
275/// Multisig] i.e. `flag || bcs_bytes(Multisig)`.
276impl ToFromBytes for GenericSignature {
277    fn from_bytes(bytes: &[u8]) -> Result<Self, RebasedError> {
278        match SignatureScheme::from_flag_byte(bytes.first().ok_or(RebasedError::InputTooShort(0))?) {
279            Ok(x) => match x {
280                SignatureScheme::ED25519 => {
281                    //| SignatureScheme::Secp256k1 | SignatureScheme::Secp256r1 => {
282                    Ok(GenericSignature::Signature(
283                        Signature::from_bytes(bytes).map_err(|_| RebasedError::InvalidSignature)?,
284                    ))
285                } // SignatureScheme::MultiSig => Ok(GenericSignature::MultiSig(MultiSig::from_bytes(bytes)?)),
286                  // SignatureScheme::ZkLoginAuthenticator => {
287                  //     let zk_login = ZkLoginAuthenticator::from_bytes(bytes)?;
288                  //     Ok(GenericSignature::ZkLoginAuthenticator(zk_login))
289                  // }
290                  // SignatureScheme::PasskeyAuthenticator => {
291                  //     let passkey = PasskeyAuthenticator::from_bytes(bytes)?;
292                  //     Ok(GenericSignature::PasskeyAuthenticator(passkey))
293                  // }
294                  // _ => Err(FastCryptoError::InvalidInput),
295            },
296            Err(_) => Err(RebasedError::InvalidInput),
297        }
298    }
299}

313impl ::serde::Serialize for GenericSignature {
314    fn serialize<S: ::serde::Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
315        if serializer.is_human_readable() {
316            #[derive(serde::Serialize)]
317            struct GenericSignature(String);
318            GenericSignature(self.encode_base64()).serialize(serializer)
319        } else {
320            #[derive(serde::Serialize)]
321            struct GenericSignature<'a>(&'a [u8]);
322            GenericSignature(self.as_ref()).serialize(serializer)
323        }
324    }
325}

327impl<'de> ::serde::Deserialize<'de> for GenericSignature {
328    fn deserialize<D: ::serde::Deserializer<'de>>(deserializer: D) -> Result<Self, D::Error> {
329        use serde::de::Error;
330
331        if deserializer.is_human_readable() {
332            #[derive(serde::Deserialize)]
333            struct GenericSignature(String);
334            let s = GenericSignature::deserialize(deserializer)?;
335            Self::decode_base64(&s.0).map_err(::serde::de::Error::custom)
336        } else {
337            #[derive(serde::Deserialize)]
338            struct GenericSignature(Vec<u8>);
339
340            let data = GenericSignature::deserialize(deserializer)?;
341            Self::from_bytes(&data.0).map_err(|e| Error::custom(e.to_string()))
342        }
343    }
