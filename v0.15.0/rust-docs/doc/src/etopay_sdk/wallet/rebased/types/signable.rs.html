8/// Something that we know how to hash and sign.
9pub trait Signable<W> {
10    fn write(&self, writer: &mut W);
11}
12
13/// Activate the blanket implementation of `Signable` based on serde and BCS.
14/// * We use `serde_name` to extract a seed from the name of structs and enums.
15/// * We use `BCS` to generate canonical bytes suitable for hashing and signing.
16///
17/// # Safety
18/// We protect the access to this marker trait through a "sealed trait" pattern:
19/// impls must be add added here (nowehre else) which lets us note those impls
20/// MUST be on types that comply with the `serde_name` machinery
21/// for the below implementations not to panic. One way to check they work is to
22/// write a unit test for serialization to / deserialization from signable
23/// bytes.
24mod bcs_signable {
25
26    pub trait BcsSignable: serde::Serialize + serde::de::DeserializeOwned {}
27    // impl BcsSignable for crate::committee::Committee {}
28    // impl BcsSignable for crate::messages_checkpoint::CheckpointSummary {}
29    // impl BcsSignable for crate::messages_checkpoint::CheckpointContents {}
30    //
31    // impl BcsSignable for crate::effects::TransactionEffects {}
32    // impl BcsSignable for crate::effects::TransactionEvents {}
33    impl BcsSignable for super::super::TransactionData {}
34    // impl BcsSignable for crate::transaction::SenderSignedData {}
35    // impl BcsSignable for crate::object::ObjectInner {}
36    //
37    // impl BcsSignable for crate::accumulator::Accumulator {}
38    //
39    // impl BcsSignable for super::bcs_signable_test::Foo {}
40    // #[cfg(test)]
41    // impl BcsSignable for super::bcs_signable_test::Bar {}
42}
43
44impl<T, W> Signable<W> for T
45where
46    T: bcs_signable::BcsSignable,
47    W: std::io::Write,
48{
49    #[allow(clippy::expect_used, reason = "invariant guaranteed by the sealed trait above")]
50    fn write(&self, writer: &mut W) {
51        let name = serde_name::trace_name::<Self>().expect("Self must be a struct or an enum");
52        // Note: This assumes that names never contain the separator `::`.
53        write!(writer, "{}::", name).expect("Hasher should not fail");
54        bcs::serialize_into(writer, &self).expect("Message serialization should not fail");
55    }
56}

58fn hash<S: Signable<H>, H: HashFunction<DIGEST_SIZE>, const DIGEST_SIZE: usize>(signable: &S) -> [u8; DIGEST_SIZE] {
59    let mut digest = H::default();
60    signable.write(&mut digest);
61    let hash = digest.finalize();
62    hash.into()
63}
64
65pub fn default_hash<S: Signable<Blake2b256>>(signable: &S) -> [u8; 32] {
66    hash::<S, Blake2b256, 32>(signable)
