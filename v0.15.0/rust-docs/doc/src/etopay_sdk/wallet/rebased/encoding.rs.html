5//! Encodings of binary data such as Base64 and Hex.
6
7use base64ct::Encoding as _;
8use serde::{Deserialize, Deserializer, Serialize, Serializer};
9use serde_with::{DeserializeAs, SerializeAs};
10
11use super::{RebasedError, traits::ToFromBytes};

13/// Trait representing a general binary-to-string encoding.
14pub trait Encoding {
15    /// Decode this encoding into bytes.
16    fn decode(s: &str) -> Result<Vec<u8>, RebasedError>;
17
18    /// Encode bytes into a string.
19    fn encode<T: AsRef<[u8]>>(data: T) -> String;
20}
21
22/// Cryptographic material with an immediate conversion to/from Base64 strings.
23///
24/// This is an [extension trait](https://rust-lang.github.io/rfcs/0445-extension-trait-conventions.html) of `ToFromBytes` above.
25///
26pub trait EncodeDecodeBase64: Sized {
27    fn encode_base64(&self) -> String;
28    fn decode_base64(value: &str) -> Result<Self, RebasedError>;
29}
30
31impl<T: ToFromBytes> EncodeDecodeBase64 for T {
32    fn encode_base64(&self) -> String {
33        Base64::encode(self.as_bytes())
34    }
35
36    fn decode_base64(value: &str) -> Result<Self, RebasedError> {
37        let bytes = Base64::decode(value)?;
38        <T as ToFromBytes>::from_bytes(&bytes)
39    }
40}

42/// Implement `DeserializeAs<Vec<u8>>`, `DeserializeAs<[u8; N]>` and `SerializeAs<T: AsRef<[u8]>`
43/// for a type that implements `Encoding`.
44macro_rules! impl_serde_as_for_encoding {
45    ($encoding:ty) => {
46        impl<'de> DeserializeAs<'de, Vec<u8>> for $encoding {
47            fn deserialize_as<D>(deserializer: D) -> Result<Vec<u8>, D::Error>
48            where
49                D: Deserializer<'de>,
50            {
51                let s = String::deserialize(deserializer)?;
52                Self::decode(&s).map_err(|_| serde::de::Error::custom("Deserialization failed"))
53            }
54        }
55
56        impl<T> SerializeAs<T> for $encoding
57        where
58            T: AsRef<[u8]>,
59        {
60            fn serialize_as<S>(value: &T, serializer: S) -> Result<S::Ok, S::Error>
61            where
62                S: Serializer,
63            {
64                let encoded_string = Self::encode(value);
65                Self(encoded_string).serialize(serializer)
66            }
67        }
68
69        impl<'de, const N: usize> DeserializeAs<'de, [u8; N]> for $encoding {
70            fn deserialize_as<D>(deserializer: D) -> Result<[u8; N], D::Error>
71            where
72                D: Deserializer<'de>,
73            {
74                let value: Vec<u8> = <$encoding>::deserialize_as(deserializer)?;
75                value
76                    .try_into()
77                    .map_err(|_| serde::de::Error::custom(format!("Invalid array length, expecting {}", N)))
78            }
79        }
80    };
81}
82
83/// Implement `TryFrom<String>` for a type that implements `Encoding`.
84macro_rules! impl_try_from_string {
85    ($encoding:ty) => {
86        impl TryFrom<String> for $encoding {
87            type Error = RebasedError;
88            fn try_from(value: String) -> Result<Self, Self::Error> {
89                // Error on invalid encoding
90                <$encoding>::decode(&value)?;
91                Ok(Self(value))
92            }
93        }
94    };
95}

97/// Base64 encoding
98#[derive(Serialize, Deserialize, Debug, Clone, Eq, PartialEq)]
99#[serde(try_from = "String")]
100pub struct Base64(String);
101
102impl Base64 {
103    /// Encodes bytes as a Base64.
104    pub fn from_bytes(bytes: &[u8]) -> Self {
105        Self(Self::encode(bytes))
106    }
107}
108
109impl_serde_as_for_encoding!(Base64);
110impl_try_from_string!(Base64);
111
112impl Encoding for Base64 {
113    fn decode(s: &str) -> Result<Vec<u8>, RebasedError> {
114        Ok(base64ct::Base64::decode_vec(s)?)
115    }
116
117    fn encode<T: AsRef<[u8]>>(data: T) -> String {
118        base64ct::Base64::encode_string(data.as_ref())
119    }
120}

122/// Hex string encoding.
123#[derive(Deserialize, Debug, Clone, PartialEq)]
124#[serde(try_from = "String")]
125pub struct Hex(String);
126
127impl Hex {
128    /// Get a string representation of this Hex encoding with a "0x" prefix.
129    pub fn encoded_with_format(&self) -> String {
130        Self::format(&self.0)
131    }
132    /// Add "0x" prefix to a hex string.
133    fn format(hex_string: &str) -> String {
134        format!("0x{}", hex_string)
135    }
136}
137
138impl TryFrom<String> for Hex {
139    type Error = RebasedError;
140    fn try_from(value: String) -> Result<Self, Self::Error> {
141        let s = value.strip_prefix("0x").unwrap_or(&value);
142        Ok(Self(s.to_string()))
143    }
144}
145
146impl Serialize for Hex {
147    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
148    where
149        S: Serializer,
150    {
151        // Hex strings are serialized with a 0x prefix which differs from the output of `Hex::encode`.
152        String::serialize(&self.encoded_with_format(), serializer)
153    }
154}
155
156impl_serde_as_for_encoding!(Hex);
157
158impl Encoding for Hex {
159    /// Decodes a hex string to bytes. Both upper and lower case characters are accepted, and the
160    /// string may have a "0x" prefix or not.
161    fn decode(s: &str) -> Result<Vec<u8>, RebasedError> {
162        let s = s.strip_prefix("0x").unwrap_or(s);
163        Ok(hex::decode(s)?)
164    }
165
166    /// Hex encoding is without "0x" prefix. See `Hex::encode_with_format` for encoding with "0x".
167    fn encode<T: AsRef<[u8]>>(data: T) -> String {
168        hex::encode(data.as_ref())
169    }
170}

172#[derive(Serialize, Deserialize, Debug, Clone, Eq, PartialEq)]
173#[serde(try_from = "String")]
174pub struct Base58(String);
175
176impl_serde_as_for_encoding!(Base58);
177impl_try_from_string!(Base58);
178
179impl Encoding for Base58 {
180    fn decode(s: &str) -> Result<Vec<u8>, RebasedError> {
181        Ok(bs58::decode(s).into_vec()?)
182    }
183
184    fn encode<T: AsRef<[u8]>>(data: T) -> String {
185        bs58::encode(data).into_string()
186    }
