11use super::super::encoding::Base64;

14/// Provides methods for executing and testing transactions.
15#[rpc(client, namespace = "iota")]
16pub trait WriteApi {
17    /// Execute the transaction and wait for results if desired.
18    /// Request types:
19    /// 1. WaitForEffectsCert: waits for TransactionEffectsCert and then return to client.
20    ///     This mode is a proxy for transaction finality.
21    /// 2. WaitForLocalExecution: waits for TransactionEffectsCert and make sure the node
22    ///     executed the transaction locally before returning the client. The local execution
23    ///     makes sure this node is aware of this transaction when client fires subsequent queries.
24    ///     However if the node fails to execute the transaction locally in a timely manner,
25    ///     a bool type in the response is set to false to indicated the case.
26    /// request_type is default to be `WaitForEffectsCert` unless options.show_events or options.show_effects is true
27    #[rustfmt::skip]
28    #[method(name = "executeTransactionBlock")]
29    async fn execute_transaction_block(
30        &self,
31        // BCS serialized transaction data bytes without its type tag, as base-64 encoded string.
32        tx_bytes: Base64,
33        // A list of signatures (`flag || signature || pubkey` bytes, as base-64 encoded string). Signature is committed to the intent message of the transaction data, as base-64 encoded string.
34        signatures: Vec<Base64>,
35        // options for specifying the content to be returned
36        options: Option<IotaTransactionBlockResponseOptions>,
37        // The request type, derived from `IotaTransactionBlockResponseOptions` if None
38        request_type: Option<ExecuteTransactionRequestType>,
39    ) -> RpcResult<IotaTransactionBlockResponse>;
40}
41
42#[derive(Serialize, Deserialize, Clone, Debug)]
43pub enum ExecuteTransactionRequestType {
44    WaitForEffectsCert,
45    WaitForLocalExecution,
46}
